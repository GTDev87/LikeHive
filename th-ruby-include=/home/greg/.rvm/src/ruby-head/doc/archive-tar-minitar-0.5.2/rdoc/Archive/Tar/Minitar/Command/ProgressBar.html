def initialize (title, total, out = STDERR)
  @title = title
  @total = total
  @out = out
  @bar_width = 80
  @bar_mark = "o"
  @current = 0
  @previous = 0
  @is_finished = false
  @start_time = Time.now
  @previous_time = @start_time
  @title_width = 14
  @format = "%-#{@title_width}s %3d%% %s %s"
  @format_arguments = [:title, :percentage, :bar, :stat]
  show
end

def inc (step = 1)
  @current += step
  @current = @total if @current > @total
  show_progress
  @previous = @current
end

def set (count)
  if count < 0 || count > @total
    raise "invalid count: #{count} (total: #{@total})"
  end
  @current = count
  show_progress
  @previous = @current
end
